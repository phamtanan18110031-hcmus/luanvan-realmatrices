    topsep     = 3pt,         % khoảng cách dọc trước và sau toàn bộ danh sách

% --- enumerate: thụt vào 2em, nhãn kiểu (1), (2), (3) — theo phong cách Słowik ---
%
%      Cú pháp label:
%        (\arabic*) → (1), (2), (3)...

%        (\roman*)  → (i), (ii), (iii)...
%

    topsep     = 3pt,
    itemsep    = 2pt,

}

\theoremstyle{plain}                         % phần thân in nghiêng (italic)
\newtheorem{thm}{Định lý}[chapter]           % Định lý 1.1, 2.3, ...
\newtheorem{prop}[thm]{Mệnh đề}             % Mệnh đề 1.2  (nối tiếp sau Định lý 1.1)
\newtheorem{lem}[thm]{Bổ đề}               % Bổ đề 1.3
\newtheorem{cor}[thm]{Hệ quả}              % Hệ quả 1.4
\newtheorem{prob}[thm]{Bài toán}            % Đã đổi từ problem thành prob để đồng bộ hoàn toàn với cleveref

\theoremstyle{definition}                    % phần thân không in nghiêng
\newtheorem{defn}[thm]{Định nghĩa}
\newtheorem{defnot}[thm]{Định nghĩa và Ký hiệu}
\newtheorem{example}{Ví dụ}[chapter]        % đánh số riêng: Ví dụ 2.1, 2.2, ...
\newtheorem{notation}{Ký hiệu}[chapter]

\theoremstyle{remark}                        % không in nghiêng, không in đậm đầu mục
\newtheorem{rem}{Nhận xét}
\newtheorem*{prob*}{Đặt vấn đề}

% Định nghĩa một môi trường riêng cho bài báo Toán (tùy chọn để tái sử dụng)
\newlist{thmenum}{enumerate}{1}
\setlist[thmenum]{
    label=(\arabic*),      % Chuyển nhãn thành (1), (2),...
    labelindent=\parindent, % Thụt lề của nhãn bằng với thụt đầu dòng của đoạn văn
    leftmargin=*,          % Tự động tính toán lề trái dựa trên nhãn
    align=left,            % Căn lề trái cho nhãn
    topsep=0.5ex,          % Khoảng cách nhỏ với đoạn văn phía trên
    itemsep=0ex            % Khoảng cách giữa các mục cực kỳ khít nhau
}
